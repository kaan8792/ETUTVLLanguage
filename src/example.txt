PROGRAM xorxnor;

DECLARATION SECTION
-- Declare variables
P, Q, R;

INITIALIZATION SECTION
-- Initialize the variables
Q=TRUE;
P=UNKNOWN;
R=FALSE;

MAIN SECTION
-- Compute the XOR of P and Q
INPUT 'Enter P:' P;
INPUT 'Enter Q:' Q;
R = NOT (P AND Q OR NOT P AND NOT Q);
OUTPUT 'P XOR Q is:' R;
OUTPUT 'P XNOR Q is:' ((P AND Q) OR (NOT P AND NOT Q));

--Test
OUTPUT 'TRUE OR TRUE=' 		(TRUE OR TRUE);
OUTPUT 'TRUE OR FALSE=' 	(TRUE OR FALSE);
OUTPUT 'FALSE OR FALSE=' 	(FALSE OR FALSE);
OUTPUT 'UNKNOWN OR TRUE=' 	(UNKNOWN OR TRUE);
OUTPUT 'UNKNOWN OR FALSE='	(UNKNOWN OR FALSE);
OUTPUT 'UNKNOWN OR UNKNOWN='(UNKNOWN OR UNKNOWN);

OUTPUT 'TRUE AND TRUE=' 		(TRUE AND TRUE);
OUTPUT 'TRUE AND FALSE=' 		(TRUE AND FALSE);
OUTPUT 'FALSE AND FALSE=' 		(FALSE AND FALSE);
OUTPUT 'UNKNOWN AND TRUE=' 		(UNKNOWN AND TRUE);
OUTPUT 'UNKNOWN AND FALSE='		(UNKNOWN AND FALSE);
OUTPUT 'UNKNOWN AND UNKNOWN='	(UNKNOWN AND UNKNOWN);
